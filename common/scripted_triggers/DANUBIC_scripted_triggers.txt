
DANUBIC_is_danubic = {
	OR = {
		original_tag = CZE
		original_tag = AUS
		original_tag = HUN
		original_tag = ROM
		original_tag = YUG
	}
}

DANUBIC_is_democratic = {
	AND = {
		OR = {
			original_tag = CZE
			original_tag = AUS
			original_tag = HUN
			original_tag = ROM
			original_tag = YUG
		}
		OR = {
			has_government = democratic
			has_government = neutrality
		}
		NOT = { has_completed_focus = ROM_balkans_dominance }
		NOT = { has_completed_focus = ROM_flexible_foreign_policy }
		NOT = { has_completed_focus = CZE_eject_the_president }
	}

}

DANUBIC_is_mon = {
	AND = {
		OR = {
			original_tag = CZE
			original_tag = AUS
			original_tag = HUN
			original_tag = ROM
			original_tag = YUG
		}
		CZE = {
			has_completed_focus = CZE_eject_the_president
		}
	}
	# Add more triggers if needed
}
DANUBIC_little_entente_viable = {
	OR = {
		original_tag = CZE
		original_tag = AUS
		original_tag = HUN
		original_tag = ROM
		original_tag = YUG
	}
	ROOT = {
		has_completed_focus = CZE_rapproachment_with_the_castle
	}
}

DANUBIC_austria_exists = {
	AUS = {
		AND = {
			NOT = {
				is_puppet = yes
			}
			NOT = {
				has_completed_focus = AUS_the_austrian_compromise
			}
			exists = yes
		}
	}
}

DANUBIC_alliance_created_already = { # Checks if any of the Nordic Alliances has been formed.  
	custom_trigger_tooltip = {
		#tooltip = NORDIC_alliance_created_already_TT
		OR = {
			has_global_flag = DANUBIC_cef_alliance_formed
			has_global_flag = DANUBIC_little_entente_alliance_formed
		}
	}
}

DANUBIC_basic_joint_trigger = {
	custom_trigger_tooltip = {
		OR = {
			has_country_flag = DANUBIC_li_leader
			has_country_flag = DANUBIC_cef_leader
			AND = {
				has_variable = global.DANUBIC_alliance_leader
				OR = {
					DANUBIC_is_democratic = yes
					DANUBIC_is_mon = yes
				}
				is_subject = no  
				is_in_faction_with = var:global.DANUBIC_alliance_leader 
			}	
		}
	}
	
}

DANUBIC_basic_available_trigger = {
	is_subject = no 
	DANUBIC_basic_joint_trigger = yes	
	any_allied_country = {
		DANUBIC_basic_joint_trigger = yes
	}
}

DANUBIC_has_started_li_focus_tree = {
	OR = {
		has_completed_focus = CZE_formalise_the_little_entente
	}
} 

DANUBIC_has_started_cea_focus_tree = {
	OR = {
		has_completed_focus = CZE_eject_the_president
	}
} 

DANUBIC_basic_joint_trigger = {
	custom_trigger_tooltip = {
			AND = {
				has_variable = global.NORDIC_alliance_leader
				DANUBIC_is_danubic = yes
				is_subject = no  
				is_in_faction_with = var:global.NORDIC_alliance_leader 
			}	
	}
	
}

DANUBIC_basic_joint_trigger_for_all = {
	custom_trigger_tooltip = {
			OR = {
				has_variable = global.DANUBIC_alliance_leader
				is_in_faction_with = var:global.DANUBIC_alliance_leader 
			}	
	}
	
}

DANUBIC_li_viable = {
	ROOT = {
		OR = {
			has_global_flag = DANUBIC_little_entente_alliance_formed
			has_dynamic_modifier = { modifier = DANUBIC_li_leader_dmod }
			has_dynamic_modifier = { modifier = DANUBIC_li_member_dmod }
			AND = {
				DANUBIC_is_democratic = yes 
				DANUBIC_alliance_created_already = no 
				#NORDIC_has_started_communist_focus_tree = no
				#NORDIC_has_started_monarchist_focus_tree = no
			}
			AND = {
				DANUBIC_has_started_li_focus_tree = yes 
				DANUBIC_alliance_created_already = no 
			}
		}
	}
}
