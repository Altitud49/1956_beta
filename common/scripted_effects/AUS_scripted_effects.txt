#
# calculations for factory output
# range -0.2 - 0.2
# increments 8
# increment 0.05
# loyalty = 35
#
# loyalty * increments / 100 = 2.8 #increment value between 0 and 8
# round_variable(2.8) = 3 #third increment to be used
# 3 * increment - range / 2 = value = -0.05 #value of the third increment
#

r56_bundeslaender_dynamic_modifier_calculation = {
	meta_effect = { # The actual meta effect. This can go anywhere you need it: in a decision, in a scripted effect, in a scripted GUI click effect, etc...
		debug = yes 
		text = {
			set_variable = {
				r56_bundeslaender_amplitude = -0.75	#on a scale from -1 to 0; 0 meaning no negative effects; 1 meaning maximum negative effects GLOBAL
			}
			if = {
				limit = { check_variable = { r56_[BNDLND]_loyalty > 0 } }
				add_to_variable = { r56_bundeslaender_val = 1 }	#if loyalty is positive add 1 (from -0.75 to 0.25 of a modifier)
			}
			set_temp_variable = {
				r56_bundeslaender_val = [LNDMAX]
			}
			multiply_temp_variable = {
				r56_bundeslaender_val = r56_bundeslaender_amplitude
			}
			set_variable = {
				[BNDLND]_dyn = r56_bundeslaender_val
			}	
			clear_variable = r56_bundeslaender_current 
		}
		BNDLND = "[?r56_bundeslaender_current.r56_bundeslaender_index]" # Gets the string for the selected bundesland
		LNDMAX = "[?r56_bundeslaender_current.r56_bundeslaender_max]"	# Gets the maximum value the modifier of the selected bundesland can be
	}
}
