
remove_suspicious_wehrmacht = {
	if = {
		limit = { has_idea = GER_suspicious_wehrmacht }
		remove_ideas = GER_suspicious_wehrmacht
	}
	if = {
		limit = { has_idea = GER_suspicious_wehrmacht_2 }
		remove_ideas = GER_suspicious_wehrmacht_2
	}
}
GER_fix_unemployment = {
	if = {
		limit = {
			has_idea = GER_slow_recovery2
		}
		remove_ideas = GER_slow_recovery2
	}
	else_if = {
		limit = {
			has_idea = GER_slow_recovery1
		}
		swap_ideas = {
			remove_idea = GER_slow_recovery1
			add_idea = GER_slow_recovery2
		}
	}	
	else_if = {
		limit = {
			OR = {
				NOT = { has_idea = GER_nazi_employment_sham }
				has_idea = GER_nazi_employment_sham
			}
		}
		swap_ideas = {
			remove_idea = GER_nazi_employment_sham
			add_idea = GER_slow_recovery1
		}
	}
}
