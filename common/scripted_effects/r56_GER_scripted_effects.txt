

GER_TUR_add_restorationist_sentiment = {
	if = {
		limit = { NOT = { has_global_flag = form_ottoman_empire_flag } }
		custom_effect_tooltip = GER_TUR_increase_restorationist_sentiment_tooltip
		add_to_variable = { GER_TUR_restorationist_sentiment_variable = GER_TUR_restorationist_sentiment_change }
		subtract_from_variable = { GER_TUR_republican_sentiment_variable = GER_TUR_restorationist_sentiment_change }

		clamp_variable = {
			var = GER_TUR_restorationist_sentiment_variable
			min = 0
			max = 100
		}
		clamp_variable = {
			var = GER_TUR_republican_sentiment_variable
			min = 0
			max = 100
		}
	}
}
