

GER_add_communist_influence = {
	if = {
		limit = { has_idea = GER_political_disarray }
		custom_effect_tooltip = GER_increase_communist_influence_tooltip
		add_to_variable = { GER_communist_influence_variable = GER_communist_influence_change }
		subtract_from_variable = { GER_democratic_influence_variable = GER_communist_influence_change }

		clamp_variable = {
			var = GER_communist_influence_variable
			min = 0
			max = 100
		}
		clamp_variable = {
			var = GER_democratic_influence_variable
			min = 0
			max = 100
		}
	}
}
