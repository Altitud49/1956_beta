	id = RKSE.become_reichskommisar.1
	title = RKSE.become_reichskommisar.1.t
	desc = RKSE.become_reichskommisar.1.d
	picture = GFX_report_event_generic_sign_treaty2

	fire_only_once = no

	is_triggered_only = yes

	option = { # lmao what
		name = RKSE.become_reichskommisar.1.a
		
		clr_global_flag  = establishing_RK
		
		custom_effect_tooltip = "You will remain as Germany"
	}
	option = { # Sure, why not?
		name = RKSE.become_reichskommisar.1.b
		
		hidden_effect = {
			event_target:RK_Country = {
				change_tag_from = ROOT
			}
		}
		
		clr_global_flag  = establishing_RK
	
		custom_effect_tooltip = "You will play as [RK_Country.GetName]"
        
        event_target:RK_Country = { country_event = { id = RKSE.become_reichskommisar.2 } }
	}
}


add_namespace = RKSE.reichskommisar_begging

country_event = { #small request
	id = RKSE.reichskommisar_begging.1
	title = RKSE.reichskommisar_begging.1.t
	desc = RKSE.reichskommisar_begging.1.d
	picture = GFX_report_event_generic_sign_treaty2

	fire_only_once = no

	is_triggered_only = yes

	option = { # lmao what
		name = RKSE.reichskommisar_begging.1.a
		
		clr_global_flag  = RK_begging_GER
        
        FROM = { country_event = { id = RKSE.reichskommisar_begging.9 } }
        
        ai_chance = { base = 40 }
	}
	option = { # We'll help
		name = RKSE.reichskommisar_begging.1.b
        
		add_manpower = -5000
        FROM = { add_manpower = 5000 }
        
		send_equipment = {
            equipment = infantry_equipment
            amount = 500
            target = event_target:RK_Country
        }
		send_equipment = {
            equipment = support_equipment
            amount = 10
            target = event_target:RK_Country
        }
		send_equipment = {
            equipment = artillery_equipment
            amount = 5
            target = event_target:RK_Country
        }
        
		clr_global_flag  = RK_begging_GER
        
        ai_chance = {
            base = 50
			modifier = {  
				has_equipment = {
					infantry_equipment > 2000
				}
				add = 100
			}
			modifier = { 
				has_equipment = {
					infantry_equipment < 1000
				}
				add = 0
			}
        }
	}
}
